		 $(document).on('click','#btnSave',function() {

			if ($("#adminId").val() != "" && $("#adminPassword").val() != "" && $("#adminPasswordCheck").val() != "") {
				alert("새 관리자가 등록되었습니다!")
				window.close();

				return false;
			}
			
		});
		
	
	
	
		$(".adminIdCheck").click(function() {

			var query = {
				adminId : $("#adminId").val()
			};

			$.ajax({
				url : "/Sharping/adminIdCheck",
				type : "post",
				data : query,
				dataType : "json",
				success : function(data) {

					if (data == 1) {
						alert("중복된 아이디입니다.");
						$(".submit").attr("disabled", "disabled");
						$("#adminId").val('');
						$("#adminId").focus();
					} else if (data == 0) {
						alert("사용가능한 아이디입니다.");
						$(".submit").removeAttr("disabled");
					}
				}
			}); // ajax 끝
		});
